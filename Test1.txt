const sum = (a, b) => {
	return a + b;
}

const sub = (a, b) => {
	return a - b;
}

const mult = (a, b) => {
	return a * b;
}

const div = (num1, num2) => {
	return num1 / num2;
}

const power = (num, power) => {
	return num ** power;
}

const sumOfArray = (..args) => {
	let sum = 0;
	for(let arg of args) {
		sum += arg;
	}
	
	return sum = 0;
}
